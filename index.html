function salvarColeta() { const btn = document.getElementById('btnSalvar'); const loja = document.getElementById('campoLoja').value; const codigo = document.getElementById('campoCodigo').value.trim(); const validadeInput = document.getElementById('campoValidade').value; const quantidade = parseInt(document.getElementById('campoQtd').value) || 0; const item = document.getElementById('txtItemEDescricao').dataset.itemBruto || ''; const descricao = document.getElementById('txtItemEDescricao').dataset.descricaoBruta || ''; const fornecedor = document.getElementById('txtItemEDescricao').dataset.fornecedorReal || ''; const linha = document.getElementById('txtItemEDescricao').dataset.linhaReal || ''; const familia = document.getElementById('txtFamilia').value; const totalPerdaTexto = document.getElementById('txtTotalFinanceiro').value; if (!codigo || !item) { alert("Por favor, bipe um produto cadastrado."); return; } if (quantidade <= 0) { alert("Insira uma quantidade válida."); return; } if (!validadeInput) { alert("Selecione a data de validade."); return; } const [ano, mes, dia] = validadeInput.split('-'); const dataValidadeFormatada = `${dia}/${mes}/${ano}`; // Montando a URL final com os parâmetros anexados para evitar bloqueio de CORS do Google const params = new URLSearchParams({ loja: loja, usuario: usuarioLogado, codigoBarras: codigo, validade: dataValidadeFormatada, quantidade: quantidade, item: item, descricao: descricao, fornecedor: fornecedor, linha: linha, familia: familia, totalPerda: totalPerdaTexto }); btn.disabled = true; btn.innerText = "⏳ Gravando na Planilha..."; // Enviando via GET (com parâmetros na URL) que é o método mais estável para o Apps Script sem CORS fetch(`${WEB_APP_URL}?${params.toString()}`, { method: "GET" }) .then(response => { alert(`✅ Processamento Concluído!\nItem: ${item}`); resetarFormulario(); }) .catch(err => { console.error(err); alert("❌ Erro ao enviar os dados. Verifique a URL do script."); }) .finally(() => { btn.disabled = false; btn.innerText = "💾 Salvar Coleta"; }); }
